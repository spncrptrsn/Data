//REGION Set player ingredient flags on dialog start
// We don't have to clear existing flags, because you cannot lose levels nor unlearn recipes
// -> will have at least as many ingredients available as last time
PROC
Proc_DialogFlagSetup(_Dialog,_Speaker1,_Speaker2)
AND
DB_LV_RunnerDialog(_Dialog)
AND
DB_IsPlayer((CHARACTERGUID)_Speaker2)
AND
CharacterGetLevel((CHARACTERGUID)_Speaker2,_PlayerLevel)
THEN
ProcDeclareCounter("LV_Runner_IngredientCount");
ProcIncreaseCounter("LV_Runner_IngredientCount");
PROC_LV_Runner_SetPlayerIngredientFlags(_Dialog,(CHARACTERGUID)_Speaker1,(CHARACTERGUID)_Speaker2,_PlayerLevel);

PROC
PROC_LV_Runner_SetPlayerIngredientFlags((STRING)_Dialog,(CHARACTERGUID)_Runner,(CHARACTERGUID)_Player,(INTEGER)_PlayerLevel)
AND
DB_LV_Runner_IngredientIndex(_Flag,_IngredientString)
THEN
NOT DB_LV_Runner_IngredientIndex(_Flag,_IngredientString);

PROC
PROC_LV_Runner_SetPlayerIngredientFlags((STRING)_Dialog,(CHARACTERGUID)_Runner,(CHARACTERGUID)_Player,(INTEGER)_PlayerLevel)
AND
DB_LV_Runner_Ingredient(_Ingredient,_IngredientString,_Level)
AND
_PlayerLevel >= _Level
AND
HasRecipeUnlockedWithIngredient(_Player,_Ingredient,1)
AND
DB_GlobalCounter("LV_Runner_IngredientCount",_Index)
THEN
PROC_LV_Runner_AddIngredient(_Dialog,_Runner,_Player,_Ingredient,_IngredientString,_Index);

PROC

AND
DB_LV_Runner_TranslatedStringVar(_Index,_DialogVar)
AND
IntegertoString(_Index,_IndexStr)
AND
StringConcatenate("LV_RunnerGet_Ingredient_",_IndexStr,_IngredientAvailableFlag)
AND
ItemTemplateGetDisplayString(_Ingredient,_Handle,_RefStr)
THEN
ProcIncreaseCounter("LV_Runner_IngredientCount");
ObjectSetFlag(_Player,_IngredientAvailableFlag);
DialogSetVariableTranslatedString(_Dialog,_DialogVar,_Handle,_RefStr);
DB_LV_Runner_IngredientIndex(_IngredientAvailableFlag,_IngredientString);

PROC
Proc_DialogFlagSetup(_Dialog,_Speaker1,_Speaker2)
AND
DB_LV_RunnerDialog(_Dialog)
AND
DB_GlobalCounter("LV_Runner_IngredientCount",_Index)
AND
_Index != 1

//REGION Detect when the player requests an ingredient to be fetched
IF
ObjectFlagSet(_Flag,(CHARACTERGUID)_Runner,_ID)
AND
DB_DialogName(_Dialog,_ID)
AND
DB_LV_RunnerDialog(_Dialog)
AND
DB_DialogPlayers(_ID,_Player,_)
AND
DB_LV_Runner_IngredientIndex(_Flag,_IngredientString)
THEN
ObjectClearFlag(_Runner,_Flag);
DB_LV_Runner_Fetching(_Runner,_Player,_IngredientString);

IF
DialogEnded(_Dialog,_ID)
AND
DB_LV_RunnerDialog(_Dialog)
AND
DB_LV_Runner_Fetching(_Runner,_Player,_IngredientString)
THEN
NOT DB_LV_Runner_Fetching(_Runner,_Player,_IngredientString);
DB_LV_Runner_BroughtBack(_Runner,_Player,_IngredientString);
SetHasDialog(_Runner,0);
DB_LV_RunnerDisappearingOutOfSight(_Runner);
CharacterDisappearOutOfSight((CHARACTERGUID)_Runner,120,1,"LV_Runner_OutOfSight",0);

IF
StoryEvent((CHARACTERGUID)_Runner,"LV_Runner_OutOfSight")
THEN
NOT DB_LV_RunnerDisappearingOutOfSight(_Runner);
//END_REGION

AND
DB_LV_Runner_BroughtBack(_Runner,_Player,_IngredientString)

PROC_LV_Runner_AppearOutOfSightIfNotReturning(_Runner);

PROC_LV_Runner_AppearOutOfSightIfNotReturning((CHARACTERGUID)_Runner)

NOT DB_LV_RunnerAppearingOutOfSight(_Runner)

DB_LV_RunnerAppearOutOfSight(_Runner);

DB_LV_RunnerAppearOutOfSight(_Runner)
AND
NOT DB_LV_RunnerDisappearingOutOfSight(_Runner)
AND
DB_CurrentLevel(_Level)
AND
DB_GLO_LevelTraveler(_Level,_Runner,_Trigger,_Dialog,_State,_SourceTrigger,_CanBeDead)
AND
GetPosition(_Trigger, _X, _Y, _Z)

NOT DB_LV_RunnerAppearOutOfSight(_Runner);

IF
StoryEvent((CHARACTERGUID)_Runner,"LV_Runner_BackAtStartPos")
THEN
NOT DB_LV_RunnerAppearingOutOfSight(_Runner);
SetHasDialog(_Runner,1);
//END_REGION
