THEN

AND
DB_RC_GY_MemorialTombs(_Tomb,_,_Dialog)
THEN
Proc_StartDialog(0,_Dialog,_Tomb,_Player);

AND
DB_RC_GY_MemorialTombs_Secrets(_Tomb,_Secret,_Reward,_Journal)
AND
ObjectGetFlag(_Player,_Journal,0)
AND
ObjectIsOnStage(_Reward,0)
THEN
AddSecret(_Player,_Secret);
Proc_RC_GY_MemorialSecretReveal((ITEMGUID)_Reward);

PROC
Proc_RC_GY_MemorialSecretReveal((ITEMGUID)_Chest)
AND
DB_ShovelRewardItemAppear(_ID,_Chest,_)
AND
DB_ShovelArea(_,_ID,_Pile)
THEN
ProcSetPerceptionDifficulty(_Pile,"Trivial");

DialogEnded(_Dialog,_ID)
AND
DB_RC_GY_MemorialDialogsToTriggerHeroes(_Dialog)
AND
DB_DialogPlayers(_ID,_Player,1)
THEN
NOT DB_RC_GY_MemorialDialogsToTriggerHeroes(_Dialog);
Proc_RC_GY_TryStartHeroRise((CHARACTERGUID)_Player);

PROC
Proc_RC_GY_TryStartHeroRise((CHARACTERGUID)_Player)
AND
NOT DB_RC_GY_MemorialDialogsToTriggerHeroes(_)
AND
QueryOnlyOnce("RC_GY_Memorial_HeroesRise")
THEN
Proc_StartDialog(0,"RC_GY_Memorial",
	ITEMGUID_S_RC_GY_Memorial_HumanHeroTomb_06aa38eb-9229-417c-8e3f-8fd5e79aafc2,
	ITEMGUID_S_RC_GY_Memorial_LizardHeroTomb_d3441698-63fd-4405-91fd-9fec449c2365,
	ITEMGUID_S_RC_GY_Memorial_ElvenHeroTomb_a73db6bb-fd89-4c49-bad7-b248db1e506b,
	ITEMGUID_S_RC_GY_Memorial_DwarvenHeroTomb_f2161dad-8cd9-4143-880e-668ee0ce4c97,
	_Player
);

IF
CharacterUsedItem(_Player,ITEMGUID_S_RC_GY_MemorialOfHeroes_9503c708-09a4-49ab-8c20-7827de172eea)
AND
GlobalGetFlag("RC_GY_MemorialSpawn",1)
THEN
StartVoiceBark("RC_GY_VB_Memorial",_Player);

IF
DialogRequestFailed("RC_GY_Memorial",_)
THEN
Proc_RC_GY_SpawnHeroes();

IF
DialogEnded("RC_GY_Memorial",_)
THEN
Proc_RC_GY_SpawnHeroes();

PROC
Proc_RC_GY_SpawnHeroes()
AND
QueryOnlyOnce("RC_GY_MemorialSpawn")
AND
DB_RC_GY_MemorialHero(_Hero,_)
AND
Random(3000,_Timer)
THEN
ProcObjectTimer(_Hero,"Timer_RC_GY_Memorial_FX",_Timer);

PROC
Proc_RC_GY_SpawnHeroes()
AND
DB_IsPlayer(_Player)
THEN
ProcHideMarker(_Player,"RC_GY_MemorialOfHeroes");

PROC
ProcObjectTimerFinished((CHARACTERGUID)_Hero,"Timer_RC_GY_Memorial_FX")
AND
GetPosition(_Hero,_X,_Y,_Z)
THEN
PlayEffectAtPosition("RS3_FX_GP_Combat_Death_Petrified_01_Medium",_X,_Y,_Z);
ProcObjectTimer(_Hero,"Timer_RC_GY_Memorial_Appear",1600);

PROC
ProcObjectTimerFinished((CHARACTERGUID)_Hero,"Timer_RC_GY_Memorial_Appear")
AND
DB_RC_GY_MemorialHero(_Hero,_Grave)
AND
DB_RC_GY_MemorialTombs(_Grave,_Template,_)
THEN
CharacterAppear(_Hero,1,"");
Transform(_Grave,_Template,1);
ItemSetCanInteract(_Grave,0);

IF
DB_RC_GY_MemorialHero(_Hero,_)
THEN
ApplyStatus(_Hero,"QUEST_PROMISE",-1.0,1);
