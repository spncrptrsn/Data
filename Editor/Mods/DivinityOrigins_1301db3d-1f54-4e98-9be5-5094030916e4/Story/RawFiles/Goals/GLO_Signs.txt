INITSECTION
/* DB_IsSign will fire the region's AD on Use action with the Object
DB_IsSign(_Obj,_RegionAD,_Flag); 

Region AD will have a collection of all signs and alike, which will use the Flag set on init on that Obj
*/

KBSECTION
IF
DB_IsSign(_Obj,(STRING)_,(STRING)_Flag)
THEN
ObjectSetFlag(_Obj,_Flag,0);

IF
CharacterUsedItem(_Char, _Obj)
AND
DB_IsSign(_Obj,_RegionDialog,_)
THEN
Proc_StartDialog(1,_RegionDialog, _Obj, _Char);

IF
DialogEnded("RC_DW_WantedPoster2",_Inst)
AND
DB_DialogPlayers(_ID, _Player, 1)
AND
QueryOnlyOnce("RC_DW_WantedPoster2")
THEN
StartVoiceBark("RC_DW_VB_WantedPoster2",(CHARACTERGUID)_Player);

IF
DialogEnded("RC_DW_WantedPoster3",_Inst)
AND
DialogGetInvolvedPlayer(_Inst,1,(CHARACTERGUID)_Player)
AND
QueryOnlyOnce("RC_DW_WantedPoster3")
THEN
ProcDefineReflectionDialog("RC_DW_RD_WantedPoster3",_Player);

PROC
Proc_ReflectionDialogDone("RC_DW_RD_WantedPoster3",(CHARACTERGUID)_Player)
AND
QueryOnlyOnce("RC_DW_VB_WantedPoster3")
THEN
StartVoiceBark("RC_DW_VB_WantedPoster3",_Player);

EXITSECTION
